# GlassTrax-Bridge Dependencies

# ======================
# Core Database Connectivity
# ======================
pyodbc                      # Pervasive SQL ODBC connection

# ======================
# Configuration Management

ruamel.yaml                 # YAML with comment preservation (for config editing)
python-dotenv               # Environment variable loading
tzdata                      # Timezone data for zoneinfo (required on Windows)

uvicorn[standard]           # ASGI server

# Database & ORM (App DB)
# ======================

python-multipart            # Form data parsing (file uploads)

# ======================

# ======================
